
[dependencies]:
	gcc
	make
	perl
	git
	libsdl2-dev



[git]:

setup:
	-mkdir project-name-git
	-git init
	-git config --global user.name "username"
	-git config --global user.email "email-address"
	-git config remote.origin.url https://username:personal-access-token@github.com/username/repository
	-git remote -v
	-git config --global alias.logline "log --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit"

check status:
	(check status every session)
	-git status

local <- remote:
	(start every session with pull)
	-git pull
	
	(after first pull, rename master to main)
	-git branch -m master main

local -> remote:
	(end every session with push)
	-git push
	
	(on first push, set default remote and local branches)
	-git push --set-upstream origin main

use case:
	-git pull

	(begin changes)
	-gedit filename
	-make
	-./app
	-rm app

	(when change is complete, commit before moving to next change)
	-git add file1 file2 file3
	-git commit -m "message"

	(repeat for each change. when all changes complete, push)
	-git push origin main

add all changes:
	(use sparingly)
	-git add .

view changelog:
	-git logline
